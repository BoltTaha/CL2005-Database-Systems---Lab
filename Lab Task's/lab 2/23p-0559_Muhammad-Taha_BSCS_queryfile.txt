1. SELECT FIRST_NAME, LAST_NAME FROM EMPLOYEES;
2. SELECT HIRE_DATE AS "Joining Date" FROM EMPLOYEES;
3. SELECT FIRST_NAME || ' ' || LAST_NAME AS "FULL NAME" FROM EMPLOYEES;
4. SELECT salary FROM EMPLOYEES WHERE SALARY * 12 > 20000 AND SALARY * 12 < 30000;
5. SELECT employee_id, FIRST_NAME FROM EMPLOYEES WHERE DEPARTMENT_ID BETWEEN 60 AND 100;
6. SELECT * FROM EMPLOYEES WHERE COMMISSION_PCT IS NULL;
7. SELECT * FROM EMPLOYEES ORDER BY SALARY DESC;
8. SELECT INITCAP(FIRST_NAME), LOWER(JOB_ID) FROM EMPLOYEES;
9. SELECT employee_id, HIRE_DATE, MONTHS_BETWEEN(SYSDATE, HIRE_DATE) AS Month_between, NEXT_DAY(HIRE_DATE, 'FRIDAY') AS nextday, LAST_DAY(HIRE_DATE) AS lastday FROM EMPLOYEES WHERE MONTHS_BETWEEN(SYSDATE, HIRE_DATE) > 100;
10. SELECT employee_id, HIRE_DATE, TO_CHAR(HIRE_DATE, 'MONTH') AS MONTH_STARTED FROM EMPLOYEES WHERE TO_CHAR(HIRE_DATE, 'YYYY') = '2003';
11. SELECT NEXT_DAY(SYSDATE, 'MONDAY') AS NEXT_MONDAY_DATE FROM DUAL;
12. SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE 'A%';
13. SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE '%a%';
14. SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE '%a';
15. SELECT first_name, SUBSTR(first_name, LENGTH(first_name) - 2, 3) AS substr_function FROM employees;
16. SELECT * from EMPLOYEES WHERE rownum <= 10 ORDER BY EMPLOYEE_ID DESC ;
17.SELECT department_id, COUNT(*) AS TOTAL_EMPLOYEE FROM EMPLOYEES WHERE DEPARTMENT_ID IS NOT NULL GROUP BY DEPARTMENT_ID ORDER BY TOTAL_employee desc;
18.SELECT FIRST_NAME || ' ' || LAST_NAME AS "FULL_NAME" , employee_id,salary, SALARY + (SALARY * 0.15) AS "INCREMENTED_SALARY", (SALARY + (SALARY * 0.15)) - SALARY  AS "INCREMENTED_AMOUNT" FROM EMPLOYEES;
19.SELECT manager_id, department_id, ROUND( AVG(commission_pct),2) AS "AVG_COMMISION" FROM EMPLOYEES WHERE COMMISSION_PCT IS NOT NULL GROUP BY manager_id, department_id;
